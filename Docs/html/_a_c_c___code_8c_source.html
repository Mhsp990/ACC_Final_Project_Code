90 SafeD_relD = i.Safe_distance - Relative_distance_pres;
91 Control_x = (Relative_velo * Kvx_gain) - ((i.Safe_distance - Relative_distance_pres) * Kxerr_gain);

93 // Test whether it is possible to use ACC safely
